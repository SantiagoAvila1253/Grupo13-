Gestión de asistencias (ingreso desde el Menú principal)

    mostrar lista de clases (ID | Materia | Fecha | Día | Horario | Docente)
    pedir id de clase (o volver / cerrar sesión)
    si clase válida:
        mostrar encabezado de la clase (materia, fecha, día, horario, docente)

        en_panel
        mientras en_panel:
            # TABLA principal (aplica filtros si están cargados y muestra que filtro se aplicó)
            # Columnas:
            #   LEGAJO | APELLIDO, NOMBRE | Estado actual | AJ acumuladas | AI acumuladas | ¿>25%?
            mostrar_tabla_clase(clase_id, filtro_texto_apellido, filtro_legajo, filtro_estado)

            # Totales de la clase (al pie):
            #   Total Presentes | Total AJ | Total AI
            mostrar_totales_clase(clase_id)

            opciones:
                1) Cargar asistencia secuencial (permitir elegir legajo de inicio)
                2) Modificar/Eliminar un registro (legajo + nuevo estado)
                3) Filtros (apellido / legajo / estado / limpiar)
                4) Ver historial y totales de un alumno (todas las clases)
                0) Volver (volver a la lista de clases)
                9) Cerrar sesión

            según opción:

                si "1":  # Cargar asistencia secuencial
                    ordenar alumnos por APELLIDO, NOMBRE
                    pedir legajo_inicio (vacío = desde el primero)
                    recorrer alumnos desde legajo_inicio hasta el final:
                        pedir estado (0=Presente, 1=AJ, 2=AI; Enter = no cambiar)
                        si estado en {0,1,2}:
                            guardar/actualizar asistencia (clase_id, legajo → estado)
                    al finalizar:
                        aplicar_desactivacion_por_ausentismo()  # ver lógica más abajo

                si "2":  # Modificar/Eliminar un registro
                    pedir LEGAJO
                    pedir NUEVO ESTADO:
                        - ingresar 0/1/2 para cambiar estado
                        - ingresar "E" para eliminar el registro
                        - ingresar Enter para no cambiar
                    si NUEVO ESTADO == "E":
                        si existe asistencia (clase_id, legajo):
                            eliminar registro
                        si no existe:
                            informar "no había registro"
                    si NUEVO ESTADO en {0,1,2}:
                        guardar/actualizar asistencia (clase_id, legajo → estado)
                        aplicar_desactivacion_por_ausentismo(legajo)

                si "3":  # Filtros unificados
                    subopciones:
                        a) Filtrar por apellido (texto; vacío = sin filtro)
                        b) Filtrar por legajo (número exacto; vacío = sin filtro)
                        c) Filtrar por estado (0/1/2; vacío = sin filtro)
                        d) Limpiar todos los filtros
                        0) Volver
                    según subopción:
                        a) pedir texto → filtro_texto_apellido = texto (comparación "contiene", sin mayúsc/minúsc)
                        b) pedir legajo → filtro_legajo = valor (si vacío, sin filtro)
                        c) pedir estado → filtro_estado = 0/1/2 (si vacío, sin filtro)
                        d) poner: filtro_texto_apellido="", filtro_legajo="", filtro_estado=""
                        0) sin cambios

                si "7":  # Historial y totales de un alumno (todas las clases)
                    pedir LEGAJO
                    mostrar historial:
                        recorrer todas las clases:
                            si existe asistencia (clase_id, legajo):
                                mostrar: FECHA | MATERIA | ESTADO
                    mostrar totales del alumno (Presentes, AJ, AI, Total)
                    mostrar %Asistencia = (Presentes/Total)*100  (si Total=0, mostrar 0%)
                    mostrar %Ausentismo (AI) = (AI/Total)*100    (si Total=0, mostrar 0%)
                    ofrecer modificación puntual:
                        pedir CLASE_ID
                        pedir nuevo estado (0/1/2; o "E" para eliminar; Enter = sin cambios)
                        aplicar si corresponde y luego:
                            aplicar_desactivacion_por_ausentismo(legajo)

                si "0":
                    en_panel = FALSO  # volver a la lista de clases

                si "9":
                    devolver "logout"  # cerrar sesión
